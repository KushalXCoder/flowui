### Create Navbar Component

Create the component `navbar.tsx` in your components directory and copy the following code:

```tsx title="components/ui/navbar.tsx"
import { cn } from "@/lib/utils";
import { ComponentPropsWithoutRef, ElementType, ReactNode } from "react";

// Polymorphic component types
type CommonProps = {
    children?: ReactNode;
    className?: string;
};

type PolymorphicProps<T extends ElementType> = {
    as?: T;
} & CommonProps & Omit<ComponentPropsWithoutRef<T>, "as" | "className" | "children">;

export const Navbar = <T extends ElementType = "nav">({
    children,
    className,
    as,
    ...props
}: PolymorphicProps<T>) => {
    const Component = as || "nav";
    return (
        <Component
            className={cn("flex justify-between items-center", className)}
            {...props}
        >
            {children}
        </Component>
    )
}

export const NavbarLogo = <T extends ElementType = "a">({
    children,
    className,
    as,
    ...props
}: PolymorphicProps<T>) => {
    const Component = as || "a";
    return (
        <Component
            className={cn("flex items-center gap-2", className)}
            {...props}
        >
            {children}
        </Component>
    )
}

export const NavbarLogoText = <T extends ElementType = "span">({
    children,
    className,
    as,
    ...props
}: PolymorphicProps<T>) => {
    const Component = as || "span";
    return (
        <Component
            className={cn("text-sm", className)}
            {...props}
        >
            {children}
        </Component>
    )
}

export const NavbarContent = <T extends ElementType = "ul">({
    children,
    className,
    as,
    ...props
}: PolymorphicProps<T>) => {
    const Component = as || "ul";
    return (
        <Component
            className={cn("flex items-center gap-4", className)}
            {...props}
        >
            {children}
        </Component>
    )
}

export const NavbarItem = <T extends ElementType = "li">({
    children,
    className,
    as,
    ...props
}: PolymorphicProps<T>) => {
    const Component = as || "li";
    return (
        <Component
            className={cn("text-sm", className)}
            {...props}
        >
            {children}
        </Component>
    )
}
```

## Usage

### Basic Structure

The `Navbar` follows a compound component pattern, allowing you to compose your header exactly how you need it.

```tsx
import { 
    Navbar, 
    NavbarLogo, 
    NavbarLogoText, 
    NavbarContent, 
    NavbarItem 
} from "@/components/ui/navbar";

export function Header() {
    return (
        <Navbar>
            <NavbarLogo href="/">
                <LogoIcon />
                <NavbarLogoText>BrandName</NavbarLogoText>
            </NavbarLogo>
            
            <NavbarContent>
                <NavbarItem><a href="/docs">Docs</a></NavbarItem>
                <NavbarItem><a href="/blog">Blog</a></NavbarItem>
            </NavbarContent>
        </Navbar>
    )
}
```

## Polymorphism

All navbar components support the `as` prop, allowing you to change the underlying HTML element while maintaining the same styles.

```tsx
{/* Render as a div instead of a nav */}
<Navbar as="div" className="...">
    {/* Render as a button instead of an anchor */}
    <NavbarLogo as="button" onClick={() => ...}>
        Logo
    </NavbarLogo>
    
    {/* Render as a div instead of a list item */}
    <NavbarContent as="div">
        <NavbarItem as="span">Item</NavbarItem>
    </NavbarContent>
</Navbar>
```

## Props

### Common Props
These props are available on all components: `Navbar`, `NavbarLogo`, `NavbarLogoText`, `NavbarContent`, and `NavbarItem`.

| Prop | Type | Default | Description |
| :--- | :--- | :--- | :--- |
| `as` | `ElementType` | Varies | The HTML element or React component to render as. |
| `children` | `ReactNode` | - | The content to be rendered. |
| `className` | `string` | - | Additional CSS classes for styling. |

### Default Elements

| Component | Default Element |
| :--- | :--- |
| `Navbar` | `nav` |
| `NavbarLogo` | `a` |
| `NavbarLogoText` | `span` |
| `NavbarContent` | `ul` |
| `NavbarItem` | `li` |
